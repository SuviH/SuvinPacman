6   package suvinpacman.suvinpacman.nakyma;
7   
8   import java.awt.event.ActionListener;
9   import javax.swing.JButton;
10  import javax.swing.JPanel;
11  import suvinpacman.suvinpacman.malli.Malli;
12  
13  /**
14   *
15   * @author suvi
16   */
17  /**
18   * Aloituspaneeli, jossa peli on aloitustilassa, tasta luokasta peli alkaa ja
19   * siirtyy pelitilaan.
20   */
21  public class Aloituspaneeli extends JPanel {
22  
23      private JButton aloitaNappi;
24      private Malli malli;
25      private Nakyma nakyma;
26  
27      /**
28       * Konstruktorissa asetetaan paneeliin aloitusnappi, tapahtumakuuntelija,
29       * seka nakyma ja malli.
30       * @param malli konstruktorissa valitettava malli
31       * @param nakyma konstruktorissa valitettava nakyma
32       */
33      public Aloituspaneeli(Malli malli, Nakyma nakyma) {
34          this.malli = malli;
35          this.nakyma = nakyma;
36          this.aloitaNappi = new JButton("Aloita");
37          ActionListener bl = new ButtonListener(malli, nakyma);
38          this.aloitaNappi.addActionListener(bl);
39          this.add(aloitaNappi);
40  
41      }
42  }
