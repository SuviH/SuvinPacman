8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import java.util.logging.Level;

12  import suvinpacman.suvinpacman.malli.Malli;

15   *
16   * @author suvi

18  /**
19   * ActionListener joka hallitsee aloita-nappulan toimintoja.
20   */
21  public class ButtonListener implements ActionListener {
22  
23      private Malli malli;
24      private Nakyma nakyma;
25  
26      /**
27       * ButtonListener-luokan konstruktori.
28       *
29       * @param malli valitettava malli
30       * @param nakyma valitettava nakyma
31       */
32  
33      public ButtonListener(Malli malli, Nakyma nakyma) {
34          this.malli = malli;
35          this.nakyma = nakyma;
36      }
37  
38      @Override
39      public void actionPerformed(ActionEvent ae) {
40          malli.vaihdaPelinTila(Malli.PelinTila.PELI);
41          malli.alustaUusiPeli();
42          this.nakyma.getPelipaneeli().getNk().setPacman(malli.getKentta().getPacman());
43          nakyma.pelinTilaMuuttunut(Malli.PelinTila.PELI);
44      }
45  
46  }
